849  SUBROUTINE imslinear_ap(THIS,ICNVG,KSTP,KITER,IN_ITER, &
850  NCONV, CONVNMOD, CONVMODSTART, LOCDV, LOCDR, &
851  CACCEL, ITINNER, CONVLOCDV, CONVLOCDR, &
852  DVMAX, DRMAX, CONVDVMAX, CONVDRMAX)
853  ! -- modules
854  USE simmodule
855  ! -- dummy variables
856  CLASS(imslineardatatype), INTENT(INOUT) :: THIS
857  integer(I4B), INTENT(INOUT) :: ICNVG !<
858  integer(I4B), INTENT(IN) :: KSTP !<
859  integer(I4B), INTENT(IN) :: KITER !<
860  integer(I4B), INTENT(INOUT) :: IN_ITER !<

867  character(len=31), DIMENSION(NCONV), INTENT(INOUT) :: CACCEL !<
868  integer(I4B), DIMENSION(NCONV), INTENT(INOUT) :: ITINNER !<
869  integer(I4B), DIMENSION(CONVNMOD, NCONV), INTENT(INOUT) :: CONVLOCDV !<
870  integer(I4B), DIMENSION(CONVNMOD, NCONV), INTENT(INOUT) :: CONVLOCDR !<
871  real(DP), DIMENSION(CONVNMOD), INTENT(INOUT) :: DVMAX !<
872  real(DP), DIMENSION(CONVNMOD), INTENT(INOUT) :: DRMAX !<
873  real(DP), DIMENSION(CONVNMOD, NCONV), INTENT(INOUT) :: CONVDVMAX !<
874  real(DP), DIMENSION(CONVNMOD, NCONV), INTENT(INOUT) :: CONVDRMAX !<
875  ! -- local variables
876  integer(I4B) :: n
877  integer(I4B) :: innerit
878  integer(I4B) :: irc
879  integer(I4B) :: itmax
880  real(DP) :: tv
881  real(DP) :: rmax
882  ! -- code
883  !
884  ! -- set epfact based on timestep
885  IF (this%ICNVGOPT == 2) THEN
886  IF (kstp == 1) THEN
887  this%EPFACT = 0.01
888  ELSE
889  this%EPFACT = 0.10
890  END IF
891  ELSE IF (this%ICNVGOPT == 4) THEN
892  this%EPFACT = dem4
893  ELSE
894  this%EPFACT = done
895  END IF
896 
897 !-------SCALE PROBLEM
898  IF (this%ISCL.NE.0) THEN
899  CALL imslinearsub_scale(0,this%ISCL, &
900  this%NEQ,this%NJA,this%IA,this%JA, &
901  this%AMAT,this%X,this%RHS, &
902  this%DSCALE,this%DSCALE2)
903  END IF

906  IF (this%IORD.NE.0) THEN

908  & this%ARO,this%JARO,this%IARO,this%LORDER,this%ID,1)

911  this%IA0 => this%IARO
912  this%JA0 => this%JARO
913  this%A0 => this%ARO
914  ELSE
915  this%IA0 => this%IA
916  this%JA0 => this%JA
917  this%A0 => this%AMAT
918  END IF

921  CALL imslinearsub_pcu(this%iout,this%NJA,this%NEQ,this%NIAPC,this%NJAPC, &

923  this%APC,this%IAPC,this%JAPC,this%IW,this%W, &
924  this%LEVEL, this%DROPTOL, this%NJLU, this%NJW, &

927  IF (kiter == 1 ) this%NITERC = 0
928  irc = 1
929  icnvg = 0
930  DO n = 1, this%NEQ
931  this%D(n) = dzero
932  this%P(n) = dzero
933  this%Q(n) = dzero
934  this%Z(n) = dzero
935  END DO
936 !-------CALCULATE INITIAL RESIDUAL
937  CALL imslinearsub_mv(this%NJA,this%NEQ,this%A0,this%X,this%D, &
938  this%IA0,this%JA0)
939  rmax = dzero
940  this%L2NORM0 = dzero
941  DO n = 1, this%NEQ
942  tv = this%D(n)
943  this%D(n) = this%RHS(n) - tv
944  IF (abs( this%D(n) ) > rmax ) rmax = abs( this%D(n) )
945  this%L2NORM0 = this%L2NORM0 + this%D(n) * this%D(n)
946  END DO
947  this%L2NORM0 = sqrt(this%L2NORM0)
948 !-------CHECK FOR EXACT SOLUTION
949  itmax = this%ITER1
950  IF (rmax == dzero) THEN
951  itmax = 0
952  icnvg = 1
953  END IF

955  IF (this%ILINMETH == 1) THEN

959  this%DVCLOSE, this%RCLOSE, this%L2NORM0, &

961  this%IAPC, this%JAPC, this%APC, &
962  this%X, this%RHS, this%D, this%P, this%Q, this%Z, &
963  this%NJLU, this%IW, this%JLU, &
964  nconv, convnmod, convmodstart, locdv, locdr, &
965  caccel, itinner, convlocdv, convlocdr, &
966  dvmax, drmax, convdvmax, convdrmax)
967 !-------SOLUTION BY THE BICONJUGATE GRADIENT STABILIZED METHOD
968  ELSE IF (this%ILINMETH == 2) THEN
969  CALL imslinearsub_bcgs(icnvg, itmax, innerit, &
970  this%NEQ, this%NJA, this%NIAPC, this%NJAPC, &
971  this%IPC, this%NITERC, this%ICNVGOPT, this%NORTH,&
972  this%ISCL, this%DSCALE, &
973  this%DVCLOSE, this%RCLOSE, this%L2NORM0, &
974  this%EPFACT, this%IA0, this%JA0, this%A0, &
975  this%IAPC, this%JAPC, this%APC, &
976  this%X, this%RHS, this%D, this%P, this%Q, &
977  this%T, this%V, this%DHAT, this%PHAT, this%QHAT, &
978  this%NJLU, this%IW, this%JLU, &
979  nconv, convnmod, convmodstart, locdv, locdr, &
980  caccel, itinner, convlocdv, convlocdr, &
981  dvmax, drmax, convdvmax, convdrmax)
982  END IF
983 !
984 !-------BACK PERMUTE AMAT, SOLUTION, AND RHS
985  IF (this%IORD.NE.0) THEN
986  CALL ims_dperm(this%NEQ, this%NJA, this%A0, this%JA0, this%IA0, &
987  & this%AMAT, this%JA,this%IA,this%IORDER,this%ID,1)
988  CALL ims_vperm(this%NEQ, this%X, this%IORDER)
989  CALL ims_vperm(this%NEQ, this%RHS, this%IORDER)
990  END IF
991 !
992 !-------UNSCALE PROBLEM
993  IF (this%ISCL.NE.0) THEN
994  CALL imslinearsub_scale(1, this%ISCL, &
995  this%NEQ, this%NJA, this%IA, this%JA, &
996  this%AMAT, this%X, this%RHS, &
997  this%DSCALE, this%DSCALE2)
998  END IF
999 !
1000 !-------SET IMS INNER ITERATION NUMBER (IN_ITER) TO NUMBER OF
1001 ! IMSLINEAR INNER ITERATIONS (innerit)
1002  in_iter = innerit
1003 !
1004 !-------RETURN
1005  RETURN
